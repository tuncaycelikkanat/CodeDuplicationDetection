   bool isSubsetSum(int set[], int n, int sum) { bool sub[n+1][sum+1]; for(int i=0; i<=n; i++) sub[i][0]=true; for(int i=1; i<=sum; i++) sub[0][i]=false; for(int i=1; i<=n; i++) for(int j=1; j<=sum; j++) {if(j<set[i-1]) sub[i][j]=sub[i-1][j]; else sub[i][j]=sub[i-1][j] || sub[i-1][j-set[i-1]];} return sub[n][sum]; }
