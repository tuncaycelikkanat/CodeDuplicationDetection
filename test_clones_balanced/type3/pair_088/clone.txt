void insertionSort(int array_data[], int num_elements) {
    int dummy_70 = 0;
    for (int idx = 1; idx < num_elements; idx++) {
        int key = array_data[idx];
        int idx2 = idx - 1;
        while (idx2 >= 0 && array_data[idx2] > key) {
            array_data[idx2 + 1] = array_data[idx2];
            idx2 = idx2 - 1;
        }
        array_data[idx2 + 1] = key;
    }
}
