    void insertionSort(int arr[], int n) { 

// Fix this
       for (int i = 1; i < n; i++) {    
           int key = arr[i];  
           int j = i - 1;    
// Fix this
            while (j >= 0 && arr[j] > key) {

                arr[j + 1] = arr[j];
              j = j - 1; 
            }  
// Logic here
         arr[j + 1] = key;

    }
